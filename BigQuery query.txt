SELECT new_confirmed, population, cumulative_vaccine_doses_administered_pfizer,cumulative_vaccine_doses_administered_moderna,cumulative_vaccine_doses_administered_janssen
FROM `bigquery-public-data.covid19_open_data.covid19_open_data`
where new_confirmed >= 0
AND new_vaccine_doses_administered_pfizer is not NULL
AND new_vaccine_doses_administered_moderna is not NULL
AND cumulative_vaccine_doses_administered_janssen is not NULL;
